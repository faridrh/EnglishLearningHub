        dot.classList.add('bg-slate-300');

          dot.classList.remove('bg-slate-300');

          dot.classList.add('active');

            dot.classList.remove('bg-slate-300');
            dot.classList.add('bg-cyan-500');
          }
        }
      });

    function flipCard() {
      const cardInner = document.getElementById('card-inner');
      isFlipped = !isFlipped;

    }

    function updateCard() {
      const card = flashcards[currentIndex];

      document.getElementById('card-emoji').textContent = card.emoji;
      document.getElementById('card-back-emoji').textContent = card.backEmoji;
      document.getElementById('progress-text').textContent = `${currentIndex + 1} / ${flashcards.length}`;

        isFlipped = false;
        document.getElementById('card-inner').classList.remove('flipped');
      }
      
      updateProgressDots();
    }

    function nextCard() {
      if (currentIndex < flashcards.length - 1) {
        currentIndex++;
        updateCard();
      } else {
        checkCompletion();
      }
    }

    function prevCard() {
      if (currentIndex > 0) {
        currentIndex--;
        updateCard();
      }
    }

    function markKnown() {

      learningCards.delete(currentIndex);
      updateStats();
      updateProgressDots();

      } else {
        checkCompletion();
      }
    }

    function markLearning() {

      } else {
        checkCompletion();

    }

    }

    function checkCompletion() {
      if (knownCards.size + learningCards.size === flashcards.length) {
        document.getElementById('completion-message').classList.remove('hidden');
      }
    }

    function resetDeck() {

      knownCards.clear();
