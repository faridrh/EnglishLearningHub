function checkAnswers(exerciseId) {
  const exercise = document.getElementById(exerciseId);
  const inputs = exercise.querySelectorAll('.fill-blank');
  const resultsDiv = document.getElementById(exerciseId + '-results');
  const scoreElement = document.getElementById(exerciseId + '-score');
  const feedbackDiv = document.getElementById(exerciseId + '-feedback');
  
  let correct = 0;
  let total = inputs.length;
  let feedback = '';
  
  inputs.forEach((input, index) => {
    const userAnswer = input.value.trim();
    const correctAnswer = input.getAttribute('data-answer');
    
    // Handle empty answers (where no article is needed)
    if (correctAnswer === '') {
      if (userAnswer === '') {
        input.classList.remove('incorrect');
        input.classList.add('correct');
        correct++;
      } else {
        input.classList.remove('correct');
        input.classList.add('incorrect');
        feedback += `<p><strong>Sual ${index + 1}:</strong> Düzgün cavab: (boş) - Sizin cavabınız: "${userAnswer}"</p>`;
      }
    } else {
      // Handle answers that require specific text
      if (userAnswer.toLowerCase() === correctAnswer.toLowerCase()) {
        input.classList.remove('incorrect');
        input.classList.add('correct');
        correct++;
      } else {
        input.classList.remove('correct');
        input.classList.add('incorrect');
        feedback += `<p><strong>Sual ${index + 1}:</strong> Düzgün cavab: "${correctAnswer}" - Sizin cavabınız: "${userAnswer}"</p>`;
      }
    }
  });
  
  const percentage = Math.round((correct / total) * 100);
  scoreElement.textContent = `Nəticə: ${correct}/${total} (${percentage}%)`;
  
  if (feedback === '') {
    feedbackDiv.innerHTML = `<p style="color: green;">Əla! Bütün cavablar düzgündür! 🎉</p>`;
  } else {
    feedbackDiv.innerHTML = feedback;
  }
  
  resultsDiv.style.display = 'block';
}

function resetExercise(exerciseId) {
  const exercise = document.getElementById(exerciseId);
  const inputs = exercise.querySelectorAll('.fill-blank');
  const resultsDiv = document.getElementById(exerciseId + '-results');
  
  inputs.forEach(input => {
    input.value = '';
    input.classList.remove('correct', 'incorrect');
  });
  
  resultsDiv.style.display = 'none';
}
</script>
